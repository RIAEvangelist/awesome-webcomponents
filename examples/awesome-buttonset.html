The awesome-buttonset element will trigger a change event when a button is clicked. Check the e.target.value to see which button index is currently active.

Paste the following into your console to see an example. 


            
                const buttons=document.querySelectorAll('awesome-buttonset');
                buttons.includes=Array.prototype.includes;

                document.addEventListener(
                    'change',
                    buttonsetChange
                );

                function buttonsetChange(e){
                    if(!buttons.includes(e.target)){
                        return;
                    }

                    console.log(e.target);
                    console.log(`The value is : ${e.target.value}`);
                }
